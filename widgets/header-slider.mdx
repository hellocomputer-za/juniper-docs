# Header Slider Widget Component

## Description

A responsive and interactive header slider component built with Bootstrap framework and Swiper.js. This widget provides a full-width carousel for showcasing hero content, images, and call-to-action elements. The header slider uses Bootstrap's utility classes, Swiper.js for smooth animations, and can be easily customized with SCSS variables and advanced styling options through the Juniper UI system.

## HTML Structure

The header slider widget uses a custom structure with Bootstrap classes, Swiper.js integration, and AOS animations through Nunjucks templates:

- `{{ animations.el(data.widget) }}`: Juniper UI animation element helper
- `data-header-slider-widget`: JavaScript selector for widget functionality
- `swiper`: Swiper.js container class
- `swiper-wrapper`: Wrapper for all slides
- `swiper-slide`: Individual slide container
- `header-slide`: Custom slide styling class
- `header-slide-background`: Background image container
- `header-slide-overlay`: Overlay for text readability
- `header-slide-content`: Content container with text and buttons
- `swiper-button-next/prev`: Navigation arrow buttons
- `swiper-pagination`: Pagination dots
- `header-scroll-indicator`: Custom scroll indicator element

## JavaScript Implementation

```javascript
// Initialize header slider with Swiper.js
document.addEventListener('DOMContentLoaded', function() {
  const headerSlider = document.querySelector('[data-header-slider-widget] .header-swiper');
  
  if (headerSlider) {
    const swiper = new Swiper(headerSlider, {
      // Basic settings
      loop: true,
      effect: 'fade',
      speed: 1000,
      autoplay: {
        delay: 5000,
        disableOnInteraction: false,
      },
      
      // Navigation
      navigation: {
        nextEl: '.header-swiper-button-next',
        prevEl: '.header-swiper-button-prev',
      },
      
      // Pagination
      pagination: {
        el: '.header-swiper-pagination',
        clickable: true,
        dynamicBullets: true,
      },
      
      // Responsive breakpoints
      breakpoints: {
        320: {
          slidesPerView: 1,
          spaceBetween: 0,
        },
        768: {
          slidesPerView: 1,
          spaceBetween: 0,
        },
        1024: {
          slidesPerView: 1,
          spaceBetween: 0,
        }
      },
      
      // Events
      on: {
        init: function() {
          console.log('Header slider initialized');
          animateSlideContent(0);
        },
        slideChange: function() {
          console.log('Slide changed to:', this.activeIndex);
          animateSlideContent(this.activeIndex);
        },
        slideChangeTransitionStart: function() {
          console.log('Slide transition started');
        },
        slideChangeTransitionEnd: function() {
          console.log('Slide transition ended');
        }
      }
    });
    
    // Custom slide content animation
    function animateSlideContent(slideIndex) {
      const slides = headerSlider.querySelectorAll('.swiper-slide');
      const currentSlide = slides[slideIndex];
      
      if (currentSlide) {
        const heading = currentSlide.querySelector('.header-slide-heading');
        const subheading = currentSlide.querySelector('.header-slide-subheading');
        const buttons = currentSlide.querySelector('.header-slide-buttons');
        
        // Reset animations
        [heading, subheading, buttons].forEach(el => {
          if (el) {
            el.style.opacity = '0';
            el.style.transform = 'translateY(30px)';
          }
        });
        
        // Animate in sequence
        setTimeout(() => {
          if (heading) {
            heading.style.transition = 'all 0.8s ease-out';
            heading.style.opacity = '1';
            heading.style.transform = 'translateY(0)';
          }
        }, 300);
        
        setTimeout(() => {
          if (subheading) {
            subheading.style.transition = 'all 0.8s ease-out';
            subheading.style.opacity = '1';
            subheading.style.transform = 'translateY(0)';
          }
        }, 600);
        
        setTimeout(() => {
          if (buttons) {
            buttons.style.transition = 'all 0.8s ease-out';
            buttons.style.opacity = '1';
            buttons.style.transform = 'translateY(0)';
          }
        }, 900);
      }
    }
    
    // Scroll indicator functionality
    const scrollIndicator = headerSlider.querySelector('.header-scroll-indicator');
    if (scrollIndicator) {
      scrollIndicator.addEventListener('click', function() {
        const nextSection = headerSlider.nextElementSibling;
        if (nextSection) {
          nextSection.scrollIntoView({ 
            behavior: 'smooth',
            block: 'start'
          });
        }
      });
    }
    
    // Pause autoplay on hover
    headerSlider.addEventListener('mouseenter', function() {
      swiper.autoplay.stop();
    });
    
    headerSlider.addEventListener('mouseleave', function() {
      swiper.autoplay.start();
    });
  }
});

// Swiper.js Events

Swiper triggers several events you can listen for:

- `init`: Fired when Swiper is initialized
- `slideChange`: Fired when slide changes
- `slideChangeTransitionStart`: Fired when slide transition starts
- `slideChangeTransitionEnd`: Fired when slide transition ends
- `autoplayStart`: Fired when autoplay starts
- `autoplayStop`: Fired when autoplay stops
- `touchStart`: Fired when touch/drag starts
- `touchEnd`: Fired when touch/drag ends
```

## Configuration Options

The header slider widget supports various configuration options:

### Basic Settings
- `autoplay`: Enable/disable automatic sliding
- `autoplayDelay`: Delay between slides (in milliseconds)
- `loop`: Enable/disable infinite loop
- `speed`: Transition speed (in milliseconds)
- `effect`: Transition effect type

### Navigation Options
- `showNavigation`: Show/hide navigation arrows
- `showPagination`: Show/hide pagination dots
- `showScrollIndicator`: Show/hide scroll indicator

### Responsive Settings
- `slidesPerView`: Number of slides per view
- `spaceBetween`: Space between slides
- `breakpoints`: Responsive breakpoint configurations

### Animation Settings
- `animationEffects`: AOS animation effect
- `animationDuration`: Animation duration
- `animationDelay`: Animation delay
- `animationEasing`: Animation easing function

- **Lazy Loading**: Images are loaded on demand
- **Optimized Animations**: Hardware-accelerated CSS transitions
- **Memory Management**: Proper cleanup of event listeners
